	\begin{screen}
		\begin{thm}[半群の群への拡張]\label{thm:extension_of_semigroup}
			$(a,o)$を簡約的可換半群とするとき，次を満たす群$(G,\Gamma)$が存在する:
			\begin{itemize}
				\item $a \subset G$.
				\item $\forall x,y \in a\ (\ o(x,y) = \Gamma(x,y)\ )$.
			\end{itemize}
		\end{thm}
	\end{screen}
